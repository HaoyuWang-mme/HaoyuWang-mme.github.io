### 第二步：创建客户端并发送请求

把下面这段代码复制到你的 `app.js` 或者任何你需要调用Llama4的文件里。

javascript
const OpenAI = require('openai');

// 初始化客户端
const client = new OpenAI({
  // 核心：把base_url换成千聚的接口地址
  baseURL: 'https://www.qianjuai.com/v1',
  // 这里的API Key是你在千聚平台上申请的
  apiKey: '你的千聚API Key',
});

async function callLlama4() {
  try {
    const stream = await client.chat.completions.create({
      // 模型名称：这里指定Llama4
      model: 'llama4', // 注意：请确认千聚平台上Llama4的具体模型名称
      messages: [
        { role: 'system', content: '你是一个智能助手，用中文回答用户的问题。' },
        { role: 'user', content: '你好，请用中文简单介绍Llama4是什么？' },
      ],
      // 开启流式输出，提升用户体验
      stream: true,
    });

    for await (const chunk of stream) {
      process.stdout.write(chunk.choices[0]?.delta?.content || '');
    }
    console.log('\n--- 请求完成 ---');

  } catch (error) {
    console.error('调用Llama4失败:', error);
  }
}

callLlama4();
