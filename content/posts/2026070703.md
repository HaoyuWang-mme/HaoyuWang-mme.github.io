python
import os
import json
import requests
from openai import OpenAI
from datetime import datetime

# 🧠 配置你的 API Key 和 base_url
API_KEY = "你的[千聚api中转站](https://www.qianjuai.com/)key"  # 从千聚平台获取
BASE_URL = "https://www.qianjuai.com/v1"

# 📸 模型配置字典
models = {
    "dalle": "dall-e-3",           # OpenAI 官方模型
    "midjourney": "midjourney",    # MJ 逆向接入
    "sd3": "stable-diffusion-3",   # Stability AI
    "flux": "flux-pro",            # FLUX 模型
    "kling": "kling-1.0"           # 可灵
}

def generate_image(prompt: str, model_key: str = "dalle"):
    """通用图片生成函数，通过 model_key 切换模型"""
    client = OpenAI(api_key=API_KEY, base_url=BASE_URL)
    
    try:
        response = client.images.generate(
            model=models[model_key],
            prompt=prompt,
            size="1024x1024",
            quality="standard",
            n=1
        )
        return {
            "status": "success",
            "url": response.data[0].url,
            "model": model_key,
            "timestamp": datetime.now().isoformat()
        }
    except Exception as e:
        return {
            "status": "error",
            "error": str(e),
            "model": model_key
        }

# 🚀 示例：一键生成不同模型的图片
prompt = "A cat wearing a space suit, digital art"

print("--> 调用 DALL-E 3 生成...")
print(generate_image(prompt, "dalle"))

print("\n--> 调用 Midjourney 生成...")
print(generate_image(prompt, "midjourney"))

print("\n--> 调用 Stable Diffusion 3 生成...")
print(generate_image(prompt, "sd3"))

# 💾 扩展：历史记录查询（模拟）
