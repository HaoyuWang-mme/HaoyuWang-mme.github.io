### 官方接入：Kimi K2的原始API

官方给的Node.js示例通常是这么写的（假设它们有）：

javascript
const axios = require('axios');

const API_KEY = 'your_real_kimi_key';
const BASE_URL = 'https://api.kimi.moonshot.cn/v1';

async function askKimi(prompt) {
  try {
    const response = await axios.post(`${BASE_URL}/chat/completions`, {
      model: 'kimi-k2',
      messages: [{ role: 'user', content: prompt }],
      max_tokens: 1024,
    }, {
      headers: {
        'Authorization': `Bearer ${API_KEY}`,
        'Content-Type': 'application/json',
      },
    });
    console.log('官方回复：', response.data.choices[0].message.content);
  } catch (error) {
    console.error('调用失败，可能触发了隐藏费率限制：', error.response?.data || error.message);
  }
}
