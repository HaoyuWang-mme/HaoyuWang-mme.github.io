## 接入有多简单：Java示例代码

真的只是改一个base_url、换一个Key的事。这里给你一个完整的Java示例，用的是最常用的OkHttp和JSON库：

java
import okhttp3.*;
import org.json.JSONArray;
import org.json.JSONObject;
import java.io.IOException;

public class ClaudeSonnetExample {
    private static final String API_KEY = "你的千聚API Key";
    // 关键：把默认的OpenAI或Claude地址换成千聚
    private static final String BASE_URL = "https://www.qianjuai.com/v1";
    
    // OkHttp客户端，全局单例
    private static final OkHttpClient client = new OkHttpClient.Builder()
            .connectTimeout(60, java.util.concurrent.TimeUnit.SECONDS)
            .readTimeout(120, java.util.concurrent.TimeUnit.SECONDS)
            .build();

    public static String callClaudeSonnet(String userMessage) throws IOException {
        MediaType JSON = MediaType.get("application/json; charset=utf-8");

        // 构建消息体：Claude兼容的格式（千聚自动转换）
        JSONObject body = new JSONObject();
        body.put("model", "claude-3-5-sonnet-20241022");  // 支持混合模型ID
        body.put("max_tokens", 4096);

        JSONArray messages = new JSONArray();
        JSONObject messageObj = new JSONObject();
        messageObj.put("role", "user");
        messageObj.put("content", userMessage);
        messages.put(messageObj);
        body.put("messages", messages);  // 注意：使用OpenAI兼容的messages格式

        // 构建请求
        Request request = new Request.Builder()
                .url(BASE_URL + "/chat/completions")  // 统一接入点
                .header("Authorization", "Bearer " + API_KEY)
                .header("Content-Type", "application/json")
                .post(RequestBody.create(body.toString(), JSON))
                .build();

        try (Response response = client.newCall(request).execute()) {
            if (!response.isSuccessful()) {
                throw new IOException("请求失败，状态码: " + response.code());
            }
            String responseBody = response.body().string();
            JSONObject jsonResponse = new JSONObject(responseBody);
            
            // 解析返回结果：完全OpenAI兼容格式
            JSONArray choices = jsonResponse.getJSONArray("choices");
            if (choices.length() > 0) {
                JSONObject choice = choices.getJSONObject(0);
                String content = choice.getJSONObject("message").getString("content");
                return content;
            }
            return "无输出";
        }
    }

    // 流式调用示例（Spring Boot WebFlux环境可用）
    public static void callStream(String userMessage) throws IOException {
        MediaType JSON = MediaType.get("application/json; charset=utf-8");

        JSONObject body = new JSONObject();
        body.put("model", "claude-3-5-sonnet-20241022");
        body.put("max_tokens", 4096);
        body.put("stream", true);  // 开启流式

        JSONArray messages = new JSONArray();
        JSONObject messageObj = new JSONObject();
        messageObj.put("role", "user");
        messageObj.put("content", userMessage);
        messages.put(messageObj);
        body.put("messages", messages);

        Request request = new Request.Builder()
                .url(BASE_URL + "/chat/completions")
                .header("Authorization", "Bearer " + API_KEY)
                .header("Content-Type", "application/json")
                .post(RequestBody.create(body.toString(), JSON))
                .build();

        try (Response response = client.newCall(request).execute()) {
            if (!response.isSuccessful()) {
                throw new IOException("请求失败，状态码: " + response.code());
            }
            // 流式输出逐行读取
            okio.Source source = response.body().source();
            okio.Buffer buffer = new okio.Buffer();
            while (source.read(buffer, 2048) != -1) {
                String line = buffer.readUtf8Line();
                if (line != null && !line.isEmpty()) {
                    // 处理每一行SSE数据，格式与OpenAI流式一致
                    System.out.println("收到数据: " + line);
                }
            }
        }
    }

    public static void main(String[] args) {
        try {
            String reply = callClaudeSonnet("用Java写一个反向代理的简单示例，代码要短");
            System.out.println("Claude Sonnet回复:\n" + reply);
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}
