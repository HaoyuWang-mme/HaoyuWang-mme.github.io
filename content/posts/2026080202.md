import okhttp3.OkHttpClient;
import retrofit2.Retrofit;

import java.time.Duration;
import java.util.Arrays;

public class DeepseekR1Example {
    public static void main(String[] args) {
        // 【核心改动】替换base_url为千聚聚合站地址

        String apiKey = "你的千聚API Key";

        ObjectMapper mapper = defaultObjectMapper();
        OkHttpClient client = defaultClient(apiKey, Duration.ofSeconds(120));
        OpenAiService service = new OpenAiService(client);

        ChatCompletionRequest completionRequest = ChatCompletionRequest.builder()
                .model("deepseek-chat") // 此处使用千聚提供的模型别名

                .temperature(0.7)
                .build();

        // 发送请求并获取回复
        service.createChatCompletion(completionRequest)
                .getChoices()
                .forEach(choice -> System.out.println(choice.getMessage().getContent()));
    }
}
