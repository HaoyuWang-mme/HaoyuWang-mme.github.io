## 第二步：封装一个GLM的调用函数（Flow 1）

现在我们创建一个核心文件 `glmService.js`。思路是这样的：我们不对接官方的智谱SDK，而是通过OpenAI客户端，通过修改配置来调用GLM。这恰恰是聚合平台最方便的地方。

javascript
// glmService.js
const OpenAI = require('openai');
require('dotenv').config();

// 初始化OpenAI客户端，但指向千聚的地址
const client = new OpenAI({
  apiKey: process.env.QIANJU_API_KEY,
  baseURL: process.env.QIANJU_BASE_URL,
});

/**
 * 调用GLM-4模型进行对话
 * @param {string} userMessage 用户输入的消息
 * @returns {string} 模型回复内容
 */
async function callGLM4(userMessage) {
  try {
    const response = await client.chat.completions.create({
      model: 'glm-4', // 关键在这里：直接写模型名
      messages: [
        { role: 'system', content: '你是一个由[千聚ai聚合站](https://www.qianjuai.com/)接通的GLM助手。' },
        { role: 'user', content: userMessage },
      ],
      temperature: 0.7,
      max_tokens: 2048,
    });

    return response.choices[0].message.content;
  } catch (error) {
    console.error('调用GLM-4时出错：', error.message);
    throw error;
  }
}

module.exports = { callGLM4 };


看见没？这里根本没有安装智谱的任何SDK。**因为千聚的接口是完全OpenAI兼容的，所以我们直接复用官方SDK即可，只是把`model`字段写成了`glm-4`**。

> 支持的模型名可以去千聚的文档中找到，比如`glm-4`、`glm-4-plus`、`deepseek-chat`等等。

## 第三步：实现流式输出（Streaming）——更高级的用法（Flow 2）

很多AI应用，尤其是聊天场景，需要流式输出才能有好的用户体感。同样用上面初始化的`client`，我们写一个流式版本。

javascript
// 流式调用函数
async function streamGLM4(userMessage) {
  const stream = await client.chat.completions.create({
    model: 'glm-4',
    messages: [
      { role: 'system', content: '你是一个流式输出的AI助手。' },
      { role: 'user', content: userMessage },
    ],
    stream: true, // 开启流式
  });

  // 这里返回一个异步迭代器，方便在HTTP响应里逐步推送
  return stream;
}

module.exports = { callGLM4, streamGLM4 };


流式返回的`stream`对象是一个异步可迭代对象，我们可以配合`for await...of`来逐块处理数据。

## 第四步：接入Express，做一个真实的API接口

光有函数没有服务是不行的。现在我们把这两个函数通过Express挂载成真实的HTTP接口。

安装Express：`npm install express`

创建 `server.js`：

javascript
// server.js
const express = require('express');
const { callGLM4, streamGLM4 } = require('./glmService');
const app = express();
const PORT = 3000;

app.use(express.json());

// 普通问答接口
app.post('/chat', async (req, res) => {
  const { message } = req.body;
  if (!message) return res.status(400).json({ error: '消息不可为空' });

  try {
    const reply = await callGLM4(message);
    res.json({ reply });
  } catch (error) {
    res.status(500).json({ error: '服务内部错误' });
  }
});

// 流式问答接口
app.post('/chat/stream', async (req, res) => {
  const { message } = req.body;
  if (!message) return res.status(400).json({ error: '消息不可为空' });

  // 设置SSE（Server-Sent Events）响应头
  res.setHeader('Content-Type', 'text/event-stream');
  res.setHeader('Cache-Control', 'no-cache');
  res.setHeader('Connection', 'keep-alive');

  try {
    const stream = await streamGLM4(message);
    // 逐块流式推送
    for await (const chunk of stream) {
      const content = chunk.choices[0]?.delta?.content || '';
      if (content) {
        // 按SSE格式推送给客户端
        res.write(`data: ${JSON.stringify({ content })}\n\n`);
      }
    }
    res.write('data: [DONE]\n\n');
    res.end();
  } catch (error) {
    res.write(`data: ${JSON.stringify({ error: error.message })}\n\n`);
    res.end();
  }
});

app.listen(PORT, () => {
  console.log(`GLM Node 服务已启动，监听端口：${PORT}`);
});

## 第五步：高可用负载均衡源码（附赠）

单一实例服务单点故障风险高。在生产环境中，我们需要一个负载均衡器，将请求分发到多个Node.js实例（或者多个API Key）上。

下面提供一个精简但功能完善的**基于轮询（Round-Robin）策略的Node.js负载均衡器**源码。

创建一个新文件 `loadBalancer.js`：

javascript
// loadBalancer.js
class APILoadBalancer {
  constructor(keys, baseUrl) {
    // 传入多个API Key作为“上游”节点
    this.keys = keys;
    this.baseUrl = baseUrl;
    this.currentIndex = 0;
  }

  // 通过轮询获取下一个可用Key
  getNextKey() {
    // 进行简单的索引轮询
    const key = this.keys[this.currentIndex];
    this.currentIndex = (this.currentIndex + 1) % this.keys.length;
    return key;
  }

  // 创建一个新的OpenAI客户端实例，使用某个Key
  createClient() {
    const OpenAI = require('openai');
    return new OpenAI({
      apiKey: this.getNextKey(),
      baseURL: this.baseUrl,
    });
  }
}

module.exports = APILoadBalancer;


在`glmService.js`里使用它：

javascript
// 在glmService.js中
const APILoadBalancer = require('./loadBalancer');

// 假设你有3个API Key（可以是在千聚上生成的多个独立Key）
const apiKeys = [
  process.env.QIANJU_API_KEY_1,
  process.env.QIANJU_API_KEY_2,
  process.env.QIANJU_API_KEY_3,
].filter(Boolean);

// 初始化负载均衡器
const balancer = new APILoadBalancer(apiKeys, process.env.QIANJU_BASE_URL);

async function callGLM4WithBalancer(userMessage) {
  const client = balancer.createClient(); // 每次调用获取一个不同的客户端/Key
  const response = await client.chat.completions.create({
    model: 'glm-4',
    messages: [{ role: 'user', content: userMessage }],
  });
  return response.choices[0].message.content;
}


**这个负载均衡器的好处：**

- 实现了请求的**均匀分发**，避免单个API Key因并发过大被限速。
- 如果有某个Key因为余额或网络问题暂时不可用，你可以在`getNextKey`逻辑中扩展健康检查（比如添加`try-catch`重试下一个Key）。
- 完全无状态，易于水平扩展。

你可以继续增强它：比如加入加权轮询（Key的额度不同）、一致性哈希（保证同一个会话落在同一个Key上）等。
