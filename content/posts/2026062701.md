**第二步：设置base_url和API key**

javascript
import OpenAI from "openai";

const openai = new OpenAI({
  baseURL: "https://www.qianjuai.com/v1",
  apiKey: "替换为你在千聚申请的key",
});


**第三步：发送KimiK2Thinking请求**

javascript
async function callKimiK2Thinking() {
  const completion = await openai.chat.completions.create({
    model: "kimi-k2-thinking",  // 模型名要匹配千聚上的名称
    messages: [
      { role: "user", content: "请帮我分析一下这篇文章的核心思想，并用Markdown格式输出：" },
    ],
  });
  
  console.log(completion.choices[0].message.content);
}

callKimiK2Thinking();

**如果要在Express应用中集成：**

javascript
const express = require('express');
const OpenAI = require('openai');

const app = express();
const openai = new OpenAI({
  baseURL: "https://www.qianjuai.com/v1",
  apiKey: process.env.QIANJU_API_KEY,
});

app.post('/chat', async (req, res) => {
  try {
    const response = await openai.chat.completions.create({
      model: "kimi-k2-thinking",
      messages: req.body.messages || [],
      stream: true,  // 支持流式输出
    });
    
    for await (const chunk of response) {
      res.write(chunk.choices[0]?.delta?.content || '');
    }
    res.end();
  } catch (error) {
    console.error(error);
    res.status(500).send('Error');
  }
});

app.listen(3000, () => console.log('Server running on port 3000'));
