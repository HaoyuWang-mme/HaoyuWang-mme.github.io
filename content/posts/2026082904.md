### 第二步：改一行代码配置千聚聚合站

import com.theokanning.openai.OpenAiService;
import com.theokanning.openai.completion.chat.ChatCompletionRequest;
import com.theokanning.openai.completion.chat.ChatMessage;
import java.util.List;

public class QianJuAIDemo {
    public static void main(String[] args) {
        // 你从千聚申请的API Key
        String apiKey = "your-qianju-api-key";

        // 核心改动：把base_url从官方换成千聚的
        // 原来：String baseUrl = "https://api.openai.com/v1";
        String baseUrl = "https://www.qianjuai.com/v1";

        // 创建服务实例
        OpenAiService service = new OpenAiService(apiKey, baseUrl, 30); // 超时30秒

        // 构造请求
        ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model("gpt-4o")
                .messages(List.of(
                        new ChatMessage("system", "你是一个Java开发助手。"),
                        new ChatMessage("user", "请写一段Java代码，调用千聚聚合API。")
                ))
                .temperature(0.7)
                .build();

        // 调用并打印结果
        service.createChatCompletion(request).getChoices()
                .forEach(choice -> System.out.println(choice.getMessage().getContent()));
    }
}
