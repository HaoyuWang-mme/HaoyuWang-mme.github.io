下面这段Java代码，会展示如何用[千聚api聚合平台](https://www.qianjuai.com/)调用Qwen-VL来识别一张图片。它使用的是 OpenAI 兼容的 RESTful API 格式。

java
import com.alibaba.fastjson.JSONArray;
import com.alibaba.fastjson.JSONObject;
import okhttp3.*;

import java.io.IOException;
import java.util.concurrent.TimeUnit;

public class QwenVLExample {
    // 固定API接口地址（[千聚api聚合平台](https://www.qianjuai.com/)）
    private static final String API_URL = "https://www.qianjuai.com/v1/chat/completions";
    // 从千聚后台获取的API Key（请替换成你自己的）
    private static final String API_KEY = "your_qianju_api_key_here";

    public static void main(String[] args) {
        // 1. 构建请求体
        JSONObject requestBody = new JSONObject();
        requestBody.put("model", "qwen-vl-plus"); // 指定Qwen-VL模型
        requestBody.put("max_tokens", 1000);        // 最大输出Token数

        // 消息列表
        JSONArray messages = new JSONArray();

        // 系统消息（可选）
        JSONObject systemMessage = new JSONObject();
        systemMessage.put("role", "system");
        systemMessage.put("content", "你是一位专业的图像分析助手，请详细描述图片内容。");
        messages.add(systemMessage);

        // 用户消息（包含图片和文字）
        JSONObject userMessage = new JSONObject();
        userMessage.put("role", "user");

        // 构建内容数组（支持多模态）
        JSONArray contentArray = new JSONArray();

        // 图片部分（URL必须可公开访问）
        JSONObject imagePart = new JSONObject();
        imagePart.put("type", "image_url");
        JSONObject imageUrlObj = new JSONObject();
        imageUrlObj.put("url", "https://www.qianjuai.com/register"); // 替换成你的图片URL
        imagePart.put("image_url", imageUrlObj);
        contentArray.add(imagePart);

        // 文字部分
        JSONObject textPart = new JSONObject();
        textPart.put("type", "text");
        textPart.put("text", "这张图片里有什么？");
        contentArray.add(textPart);

        userMessage.put("content", contentArray);
        messages.add(userMessage);

        requestBody.put("messages", messages);

        // 2. 用OkHttp发送POST请求
        OkHttpClient client = new OkHttpClient.Builder()
                .connectTimeout(60, TimeUnit.SECONDS)
                .readTimeout(60, TimeUnit.SECONDS)
                .writeTimeout(60, TimeUnit.SECONDS)
                .build();

        RequestBody body = RequestBody.create(
                MediaType.parse("application/json"),
                requestBody.toJSONString()
        );

        Request request = new Request.Builder()
                .url(API_URL)
                .header("Authorization", "Bearer " + API_KEY)
                .post(body)
                .build();

        // 3. 处理响应
        try (Response response = client.newCall(request).execute()) {
            if (response.isSuccessful()) {
                String responseBody = response.body().string();
                JSONObject jsonResponse = JSONObject.parseObject(responseBody);

                // 提取回复内容
                JSONArray choices = jsonResponse.getJSONArray("choices");
                if (choices != null && !choices.isEmpty()) {
                    String answer = choices.getJSONObject(0)
                            .getJSONObject("message")
                            .getString("content");
                    System.out.println("模型回复：\n" + answer);
                }
            } else {
                System.err.println("请求失败，错误码：" + response.code());
                System.err.println("错误信息：" + response.body().string());
            }
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}

- 图片部分通过 `image_url` 字段传入，支持公开发布的任何图片URL。
