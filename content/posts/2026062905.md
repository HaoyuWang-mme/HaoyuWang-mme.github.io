// 2. 构建请求体（这是省钱的核心：用对base_url）
public class Grok4Client {

    // **重点**：你只需要改这行URL和你的API Key
    private static final String API_BASE_URL = "https://www.qianjuai.com/v1";
    private static final String API_KEY = "sk-你的Key"; // 去 https://www.qianjuai.com/register 获取

    private static final HttpClient httpClient = HttpClient.newBuilder()
            .connectTimeout(Duration.ofSeconds(30))
            .build();

    // 3. 发送非流式请求
    public static String callGrok4NonStream(String userMessage) throws Exception {
        // 构建符合OpenAI标准的请求体
        String requestBody = """
                {
                    "model": "grok-4",
                    "messages": [{"role": "user", "content": "%s"}],
                    "temperature": 0.7
                }
                """.formatted(userMessage);

        HttpRequest request = HttpRequest.newBuilder()
                .uri(URI.create(API_BASE_URL + "/chat/completions")) // 关键：这里用千聚的地址

                .header("Authorization", "Bearer " + API_KEY)

                .timeout(Duration.ofSeconds(60))
                .build();

        HttpResponse<String> response = httpClient.send(request, HttpResponse.BodyHandlers.ofString());
        return response.body();
    }

            System.err.println("调用失败，请检查API Key或网络: " + e.getMessage());
        }
